  ```js
  // ── B8: Inline error helpers ─────────────────────────────────────────────
  let _lastStreamBody = null; // remembers the last POST body so Retry can re-run

  function clearError(){
    document.querySelectorAll('.error-bubble').forEach(el=>el.remove());
  }

  function showError(message, retryable){
    clearError();
    const t = document.getElementById('thread');
    const d = document.createElement('div');
    d.className = 'error-bubble';
    d.setAttribute('role','alert');
    d.innerHTML =
      '<div class="error-bubble-body">'
      +'<div class="error-bubble-title">⚠ Something went wrong</div>'
      +'<div class="error-bubble-msg">'+esc(message||'An unexpected error occurred.')+'</div>'
      +'<div class="error-bubble-actions">'
      +(retryable!==false
        ? '<button class="error-btn-retry" type="button">Retry</button>'
        : '<button class="error-btn-retry" type="button" onclick="location.href=\'/tutor/\'+SID">Reload page</button>')
      +'<button class="error-btn-dismiss" type="button">Dismiss</button>'
      +'</div>'
      +'</div>';
    t.appendChild(d);
    t.scrollTop = t.scrollHeight;
    // Wire buttons
    const retryBtn = d.querySelector('.error-btn-retry');
    if(retryBtn && retryable!==false){
      retryBtn.addEventListener('click', ()=>{
        d.remove();
        if(_lastStreamBody !== null) streamEvents(_lastStreamBody);
      });
    }
    d.querySelector('.error-btn-dismiss').addEventListener('click', ()=>{
      d.remove();
      // Re-enable input so learner can try again manually
      const inp = document.getElementById('answer-input');
      if(inp){ inp.disabled=false; inp.focus(); }
      document.getElementById('working').style.display='none';
    });
  }
  ```
